def barplota(x):
    agg_data = df.groupby(x)['cnt'].sum().reset_index()
    fig = px.bar(agg_data, x=x, y='cnt',text_auto='.2s',
                 color=x,color_continuous_scale=[[0,'rgb(0,0,255)'],[1,'rgb(255,0,0)']],barmode='stack')
    fig.update_traces(textfont_size=12, textangle=0, textposition="outside", cliponaxis=False)
    fig.update_layout(title=f" Avg Ride Count Vs {x}",title_x=0.5)
    fig.show()
In [27]:
def barplotb(x,c):
    agg_data = df.groupby([x,c])['cnt'].sum().reset_index()
    fig = px.bar(agg_data, x=x, y='cnt',text_auto='.2s',color=c, facet_col=c, barmode='stack')
    fig.update_traces(textfont_size=12, textangle=0, textposition="outside", cliponaxis=False)
    fig.update_layout(coloraxis=dict(colorscale='Viridis'))
    fig.update_layout(title=f" Avg Ride Count Vs {x} based on Year",title_x=0.5)
    fig.show()
